-- ============================================================
--	    tablas de soporte para los eventos
-- ============================================================
 
-- tabla donde registra los vencimientos de plazos detectados
CREATE TABLE IF NOT EXISTS AlertasVencimiento (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    ID_caso INT,
    Titulo_caso VARCHAR(30),
    ID_actuacion INT,
    Descripcion TEXT,
    Fecha_limite DATE,
    Dias_restantes INT,
    Fecha_alerta DATETIME
);
 
-- tabla donde guarda el reporte de horas trabajadas por abogado
CREATE TABLE IF NOT EXISTS ReporteHorasTrabajadas (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    ID_abogado INT,
    Abogado VARCHAR(80),
    Total_horas DECIMAL(8,2),
    Total_facturado DECIMAL(10,2),
    Periodo VARCHAR(25),
    Fecha_generacion DATETIME
);
 
-- tabla para registrar el monitoreo del cobro de facturas
CREATE TABLE IF NOT EXISTS MonitoreoFacturas (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    Total_facturas_pendientes INT,
    Monto_total_pendiente DECIMAL(12,2),
    Facturas_vencidas INT,
    Fecha_monitoreo DATETIME
);
 
-- tabla: análisis de carga de trabajo
CREATE TABLE IF NOT EXISTS CargaTrabajo (
    ID INT PRIMARY KEY AUTO_INCREMENT,
    ID_abogado INT,
    Abogado VARCHAR(80),
    Casos_activos INT,
    Actuaciones_pendientes INT,
    Nivel_carga VARCHAR(20),
    Fecha_analisis DATETIME
);
 
 
-- ============================================================
-- 1. EVT_VerificarVencimientosPlazos
--    verifica diariamente plazos legales próximos a vencer
-- ============================================================
 
DELIMITER $$
 
CREATE EVENT EVT_VerificarVencimientosPlazos
ON SCHEDULE EVERY 1 DAY
STARTS CURRENT_TIMESTAMP
ON COMPLETION PRESERVE
COMMENT 'verifica diariamente actuaciones con plazo próximo a vencer (7 días)'
DO
BEGIN
 
   -- limpia alertas antiguas de más de 30 días
   DELETE FROM AlertasVencimiento
   WHERE Fecha_alerta < DATE_SUB(NOW(), INTERVAL 30 DAY);
 
   -- alertas de actuaciones que vencen en los próximos 7 días
    INSERT INTO AlertasVencimiento (ID_caso, Titulo_caso, ID_actuacion, Descripcion, Fecha_limite, Dias_restantes, Fecha_alerta)
    SELECT
        c.ID,
        c.Titulo,
        a.ID,
        a.Descripcion,
        a.Fecha_limite,
        DATEDIFF(a.Fecha_limite, CURDATE()),
        NOW()
    FROM actuaciones a
    INNER JOIN casos c ON a.ID_caso = c.ID
    INNER JOIN estados_caso ec ON c.ID_estado = ec.ID
    WHERE a.Fecha_limite BETWEEN CURDATE() AND DATE_ADD(CURDATE(), INTERVAL 7 DAY)
    AND a.Resultado IS NULL
    AND ec.Estado = 'Activo'
    AND NOT EXISTS (
        SELECT 1 FROM AlertasVencimiento av
        WHERE av.ID_actuacion = a.ID
        AND DATE(av.Fecha_alerta) = CURDATE()
    );
 
END$$
 
DELIMITER ;
 
-- verific que el evento se creo
SHOW EVENTS FROM Lawfirm;
 
-- consultar las alertas generadas
SELECT * FROM AlertasVencimiento
ORDER BY Dias_restantes ASC;
 
 
-- ============================================================
-- 2. EVT_GenerarReporteHorasTrabajadas
--    genera reportes de horas trabajadas por abogado cada mes
-- ============================================================
 
DELIMITER $$
 
CREATE EVENT EVT_GenerarReporteHorasTrabajadas
ON SCHEDULE EVERY 1 MONTH
STARTS '2025-05-01 00:00:00'
ON COMPLETION PRESERVE
COMMENT 'genera mensualmente el reporte de horas trabajadas por abogado'
DO
BEGIN
    DECLARE v_periodo VARCHAR(25);
 
    -- armamos el periodo como "Mes Año" del mes anterior
    SET v_periodo = DATE_FORMAT(DATE_SUB(NOW(), INTERVAL 1 MONTH), '%M %Y');
 
    -- insertamos el reporte de horas por abogado del mes anterior
    INSERT INTO ReporteHorasTrabajadas (ID_abogado, Abogado, Total_horas, Total_facturado, Periodo, Fecha_generacion)
    SELECT
        a.ID,
        CONCAT(a.Nombres, ' ', a.Apellidos),
        COALESCE(SUM(fd.Horas), 0),
        COALESCE(SUM(fd.Horas * fd.Tarifa), 0),
        v_periodo,
        NOW()
    FROM abogados a
    LEFT JOIN factura_detalle fd ON a.ID = fd.ID_abogado
    LEFT JOIN facturas f ON fd.ID_factura = f.ID
    WHERE MONTH(f.Fecha) = MONTH(DATE_SUB(NOW(), INTERVAL 1 MONTH))
    AND YEAR(f.Fecha) = YEAR(DATE_SUB(NOW(), INTERVAL 1 MONTH))
    GROUP BY a.ID, a.Nombres, a.Apellidos;
 
END$$
 
DELIMITER ;
 
-- verificamos que el evento se creó
SHOW EVENTS FROM Lawfirm;
 
-- consultamos los reportes generados
SELECT * FROM ReporteHorasTrabajadas
ORDER BY Fecha_generacion DESC;
 
 
-- ============================================================
-- 3. EVT_ActualizarJurisprudenciaRelevante
--    actualiza la base de datos de jurisprudencia cada semana
-- ============================================================
 
DELIMITER $$
 
CREATE EVENT EVT_ActualizarJurisprudenciaRelevante
ON SCHEDULE EVERY 1 WEEK
STARTS '2025-04-28 02:00:00'
ON COMPLETION PRESERVE
COMMENT 'actualiza semanalmente la aplicabilidad de los precedentes según su uso en casos'
DO
BEGIN
 
    -- si un precedente está vinculado a más de 3 conocimientos, se marca como Alta
    UPDATE precedentes p
    SET p.Aplicabilidad = CASE
        WHEN (SELECT COUNT(*) FROM conocimiento_precedente cp WHERE cp.ID_precedente = p.ID) > 3 THEN 'Alta'
        WHEN (SELECT COUNT(*) FROM conocimiento_precedente cp WHERE cp.ID_precedente = p.ID) BETWEEN 1 AND 3 THEN 'Media'
        ELSE 'Baja'
    END;
 
    -- actualizar el resumen de los precedentes que tienen aplicabilidad Alta
    -- para reflejar que son ampliamente utilizados en el bufete
    UPDATE precedentes
    SET Resumen = CONCAT(Resumen, ' [Actualizado: ', DATE_FORMAT(NOW(), '%d/%m/%Y'), ']')
    WHERE Aplicabilidad = 'Alta'
    AND Resumen NOT LIKE '%[Actualizado%';
 
END$$
 
DELIMITER ;
 
-- verificar que el evento se creó
SHOW EVENTS FROM Lawfirm;
 
-- consultar el estado actualizado de los precedentes
SELECT ID, Tema, Instancia, Aplicabilidad
FROM precedentes
ORDER BY Aplicabilidad DESC;
 
 
-- ============================================================
-- 4. EVT_MonitorearCobroFacturas
--    monitorea el estado de cobro de facturas cada día
-- ============================================================
 
DELIMITER $$
 
CREATE EVENT EVT_MonitorearCobroFacturas
ON SCHEDULE EVERY 1 DAY
STARTS CURRENT_TIMESTAMP
ON COMPLETION PRESERVE
COMMENT 'monitorea diariamente el estado de cobro de facturas pendientes'
DO
BEGIN
 
    -- marca como vencidas las facturas pendientes con mass de 30 días sin pagar
    UPDATE facturas f
    INNER JOIN estados_pago ep ON f.ID_estado = ep.ID
    SET f.ID_estado = (SELECT ID FROM estados_pago WHERE Estado_pago = 'Vencido' LIMIT 1)
    WHERE ep.Estado_pago = 'Pendiente'
    AND f.Fecha < DATE_SUB(CURDATE(), INTERVAL 30 DAY);
 
    -- registra el monitoreo del día en la tabla de seguimiento
    INSERT INTO MonitoreoFacturas (Total_facturas_pendientes, Monto_total_pendiente, Facturas_vencidas, Fecha_monitoreo)
    SELECT
        SUM(CASE WHEN ep.Estado_pago = 'Pendiente' THEN 1 ELSE 0 END),
        SUM(CASE WHEN ep.Estado_pago = 'Pendiente' THEN f.Total ELSE 0 END),
        SUM(CASE WHEN ep.Estado_pago = 'Vencido' THEN 1 ELSE 0 END),
        NOW()
    FROM facturas f
    INNER JOIN estados_pago ep ON f.ID_estado = ep.ID;
 
END$$
 
DELIMITER ;
 
-- verifica que el evento se creó
SHOW EVENTS FROM Lawfirm;
 
-- consulta el historial de monitoreo
SELECT * FROM MonitoreoFacturas
ORDER BY Fecha_monitoreo DESC;
 
-- ves el estado actual de las facturas
SELECT ep.Estado_pago, COUNT(*) AS Total, SUM(f.Total) AS Monto
FROM facturas f
INNER JOIN estados_pago ep ON f.ID_estado = ep.ID
GROUP BY ep.Estado_pago;
 
 
-- ============================================================
-- 5. EVT_VerificarCargaTrabajo
--    analiza la carga de trabajo de los abogados cada semana
-- ============================================================
 
DELIMITER $$
 
CREATE EVENT EVT_VerificarCargaTrabajo
ON SCHEDULE EVERY 1 WEEK
STARTS CURRENT_TIMESTAMP
ON COMPLETION PRESERVE
COMMENT 'analiza semanalmente la carga de trabajo por abogado para equilibrar asignaciones'
DO
BEGIN
 
    -- limpias el análisis anterior
    DELETE FROM CargaTrabajo
    WHERE Fecha_analisis < DATE_SUB(NOW(), INTERVAL 30 DAY);
 
    -- inserta el análisis actualizado de carga por abogado
    INSERT INTO CargaTrabajo (ID_abogado, Abogado, Casos_activos, Actuaciones_pendientes, Nivel_carga, Fecha_analisis)
    SELECT
        a.ID,
        CONCAT(a.Nombres, ' ', a.Apellidos),
        COUNT(DISTINCT c.ID),
        COUNT(DISTINCT ac.ID),
        CASE
            WHEN COUNT(DISTINCT c.ID) >= 5 THEN 'Alta'
            WHEN COUNT(DISTINCT c.ID) BETWEEN 2 AND 4 THEN 'Media'
            ELSE 'Baja'
        END,
        NOW()
    FROM abogados a
    LEFT JOIN caso_abogado ca ON a.ID = ca.ID_abogado
    LEFT JOIN casos c ON ca.ID_caso = c.ID
    LEFT JOIN estados_caso ec ON c.ID_estado = ec.ID AND ec.Estado = 'Activo'
    LEFT JOIN actuaciones ac ON c.ID = ac.ID_caso AND ac.Resultado IS NULL AND ac.Fecha >= NOW()
    GROUP BY a.ID, a.Nombres, a.Apellidos;
 
END$$
 
DELIMITER ;
 
-- verifica que el evento se creó
SHOW EVENTS FROM Lawfirm;
 
-- consulta el análisis de carga de trabajo
SELECT * FROM CargaTrabajo
ORDER BY Casos_activos DESC;
 
-- un resumen del nivel de carga general del bufete
SELECT
Nivel_carga,
COUNT(*) AS Total_abogados,
AVG(Casos_activos) AS Promedio_casos,
AVG(Actuaciones_pendientes) AS Promedio_actuaciones
FROM CargaTrabajo
WHERE DATE(Fecha_analisis) = CURDATE()
GROUP BY Nivel_carga;
